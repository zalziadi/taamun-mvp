  <action>
    Create `scripts/guards/phase-11-anti-patterns.sh`. Mirror the structure of `scripts/guards/phase-09-anti-patterns.sh` — POSIX bash (`#!/usr/bin/env bash`, `set -euo pipefail`), 5 check blocks, comment-line carve-out regex, colored PASS/FAIL output, exit code accumulation.

    Pseudocode:
    ```bash
    #!/usr/bin/env bash
    set -euo pipefail

    FAIL=0
    OG_ROUTE="src/app/year-in-review/og/route.tsx"
    PHASE_GLOB=(src/app/year-in-review src/components/YearInReviewArchive.tsx src/components/AwarenessTrajectory.tsx)

    # Check 1: privacy invariant
    if grep -nE "YIRPrivateContent" "$OG_ROUTE" | grep -vE ":[[:space:]]*\*|:[[:space:]]*//"; then
      echo "FAIL: YIRPrivateContent imported into share card route"
      FAIL=1
    fi

    # Check 2: no chart libs
    if grep -nE '"(recharts|chart\.js|d3|chartist|plotly|victory)":' package.json; then
      echo "FAIL: chart library installed — Phase 11 uses hand-rolled SVG (YIR-11)"
      FAIL=1
    fi

    # Check 3: banned copy (loop over files × patterns)
    BANNED=(Wrapped ranked "top [0-9]+%" countdown scarcity "auto-share" "share automatically" streak "lost your" "achievement unlocked" "badge unlocked!")
    for path in "${PHASE_GLOB[@]}"; do
      [ -e "$path" ] || continue
      for pat in "${BANNED[@]}"; do
        if grep -rnE "$pat" "$path" | grep -vE ":[[:space:]]*\*|:[[:space:]]*//"; then
          echo "FAIL: banned vocabulary '$pat' in $path"
          FAIL=1
        fi
      done
    done

    # Check 4: edge runtime bypass in og route
    if grep -nE "supabaseAdmin|requireUser|from ['\"]next/headers['\"]|@supabase/supabase-js" "$OG_ROUTE" | grep -vE ":[[:space:]]*\*|:[[:space:]]*//"; then
      echo "FAIL: og/route.tsx touches server-only DB/auth — must be public, edge, query-param-only"
      FAIL=1
    fi

    if [ "$FAIL" -eq 0 ]; then
      echo "phase-11 anti-patterns: PASS"
      exit 0
    fi
    exit 1
    ```
