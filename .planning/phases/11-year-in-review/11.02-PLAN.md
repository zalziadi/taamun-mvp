    Create `src/lib/yearInReview/types.ts`. File header block documents:

    Exports:
    ```ts
    /**
     * Aggregate stats SAFE for outbound surfaces (share card, OG image, email).
     * Contains ZERO user-authored content — only counts, averages, day numbers,
     * ISO timestamps, and pre-aggregated number arrays.
     *
     * Shape mirrors the jsonb returned by the get_year_in_review Postgres RPC
     * (Plan 11.01). Keep in sync — if the RPC shape changes, update this type
     * and the sparkline renderer (Plan 11.05).
     */
    export interface YIRPublicStats {
      readonly reflections_count: number;
      readonly awareness_avg: number | null;
      readonly milestones_reached: readonly string[];  // badge_codes — public identifiers
      readonly cycle_count: number;
      readonly earliest_reflection_at: string | null;   // ISO 8601
      readonly latest_reflection_at: string | null;
      readonly awareness_trajectory: readonly number[]; // 0..52 weekly averages
    }

    /**
     * User-authored content — NEVER safe to ship outbound.
     * Exists as a TYPE ONLY to make accidental bleed a compile error.
     * No code should construct or pass this to any route handler or image
     * renderer. The /og/route.tsx grep guard in Plan 11.07 will fail CI if
     * this identifier appears in that file.
     */
    export interface YIRPrivateContent {
      readonly reflection_text: string;
      readonly emotion_labels: readonly string[];
      readonly guide_messages: readonly string[];
    }

    /** year_key format: "YYYY_anniversary" — computed from profiles.activation_started_at */
    export const YEAR_KEY_PATTERN = /^[0-9]{4}_anniversary$/;

    /** Runtime type guard — cheap check to rule out obviously-wrong payloads from the RPC. */
    export function isYIRPublicStats(x: unknown): x is YIRPublicStats {
      if (!x || typeof x !== "object") return false;
      const r = x as Record<string, unknown>;
      return (
        typeof r.reflections_count === "number" &&
        (typeof r.awareness_avg === "number" || r.awareness_avg === null) &&
        Array.isArray(r.milestones_reached) &&
        typeof r.cycle_count === "number" &&
        Array.isArray(r.awareness_trajectory)
      );
    }
    ```

    Constraints:
    - `readonly` on every field + `readonly` on arrays (prevents mutation after aggregation)
    - No `any`, no intersections between the two types
    - No imports — pure types, no runtime deps

    Re-run vitest — all 5 tests PASS. Commit: `feat(11-02): type-split library for YIR privacy enforcement (GREEN)`.
