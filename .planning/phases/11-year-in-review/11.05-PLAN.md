    ```tsx
    import type { YIRPublicStats } from "@/lib/yearInReview/types";

    export function AwarenessTrajectory({
      trajectory,
      width = 320,
      height = 80,
    }: {
      trajectory: YIRPublicStats["awareness_trajectory"];
      width?: number;
      height?: number;
    }) {
      if (!trajectory || trajectory.length < 2) return null;

      const min = Math.min(...trajectory);
      const max = Math.max(...trajectory);
      const range = max - min || 1;
      const stepX = width / (trajectory.length - 1);

      const points = trajectory
        .map((v, i) => {
          const x = i * stepX;
          const y = height - ((v - min) / range) * height;
          return `${x.toFixed(1)},${y.toFixed(1)}`;
        })
        .join(" ");

      return (
        <svg
          viewBox={`0 0 ${width} ${height}`}
          role="img"
          aria-label="منحنى حضور وعيك خلال السنة"
          className="w-full h-20"
        >
          <polyline
            points={points}
            fill="none"
            stroke="currentColor"
            strokeWidth={2}
            strokeLinecap="round"
            strokeLinejoin="round"
          />
        </svg>
      );
    }
    ```
    Pure function, no useEffect, no useState. Client-safe but no `"use client"` directive needed (no hooks). Commit: `feat(11-05): hand-rolled SVG awareness sparkline (GREEN)`.
