    ```ts
    export const MIN_REFLECTIONS_THRESHOLD = 30;
    export const CACHE_TTL_HOURS = 24;

    export async function getYearInReview(
      userId: string,
      supabase = supabaseAdmin,
      now = () => new Date()
    ): Promise<YIRPublicStats | null> {
      try {
        // 1. Resolve profile → year_key
        const { data: profile, error: pErr } = await supabase
          .from("profiles")
          .select("activation_started_at, created_at")
          .eq("id", userId)
          .maybeSingle();
        if (pErr || !profile) return null;
        const yearKey = yearKeyForUser(profile);

        // 2. Threshold gate — query reflections count in window
        const { start, end } = parseYearKey(yearKey, new Date(profile.activation_started_at ?? profile.created_at));
        const { count: reflectionsCount } = await supabase
          .from("reflections")
          .select("id", { count: "exact", head: true })
          .eq("user_id", userId)
          .gte("created_at", start.toISOString())
          .lt("created_at", end.toISOString());
        if ((reflectionsCount ?? 0) < MIN_REFLECTIONS_THRESHOLD) return null;

        // 3. Cache check
        const { data: cached } = await supabase
          .from("year_reviews")
          .select("payload, generated_at")
          .eq("user_id", userId)
          .eq("year_key", yearKey)
          .maybeSingle();
        if (cached && isFresh(cached.generated_at, now())) {
          return isYIRPublicStats(cached.payload) ? cached.payload : null;
        }

        // 4. Regenerate via RPC
        const { data: fresh, error: rErr } = await supabase.rpc(
          "get_year_in_review",
          { p_user_id: userId, p_year_key: yearKey }
        );
        if (rErr || !isYIRPublicStats(fresh)) {
          console.warn("[yir] RPC returned malformed payload:", rErr);
          return null;
        }

        // 5. Upsert snapshot
        await supabase
          .from("year_reviews")
          .upsert(
            { user_id: userId, year_key: yearKey, payload: fresh, generated_at: now().toISOString() },
            { onConflict: "user_id,year_key" }
          );

        return fresh;
      } catch (err) {
        console.warn("[yir] getYearInReview failed:", err);
        return null;
      }
    }

    function isFresh(generatedAt: string, now: Date): boolean {
      const ageMs = now.getTime() - new Date(generatedAt).getTime();
      return ageMs < CACHE_TTL_HOURS * 3600 * 1000;
    }
    ```
