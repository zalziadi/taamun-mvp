```js
/*
 * Phase 8 anti-pattern guard.
 *
 * Enforces banned tokens/patterns that would violate:
 *   - BADGE-04 (badges private — no share)
 *   - BADGE-09 (no modal interrupting flow)
 *   - 08-CONTEXT.md banned anti-patterns list
 *   - SUMMARY.md R4 gamification tone anti-patterns (Phase 8 subset)
 *
 * Failure exits with code 1. Wired into `guard:release` via package.json.
 * Mirrors scripts/guards/phase-7-anti-pattern.js (07.06) pattern.
 */

const fs = require("fs");
const path = require("path");
const { globSync } = require("glob");

const RULES = [
  {
    name: "BADGE-04: no share / next-og / opengraph on badge components",
    pattern: /\b(onShare|next\/og|ImageResponse|opengraph)\b|share.*button|share.*badge/i,
    paths: ["src/components/badges/**/*.{ts,tsx}", "src/app/progress/**/*.{ts,tsx}"],
  },
  {
    name: "BADGE-09: no modal/dialog/drawer/sheet on badge or progress page",
    pattern: /\b(Dialog|Modal|Drawer|Sheet)\b/,
    paths: ["src/components/badges/**/*.{ts,tsx}", "src/app/progress/**/*.{ts,tsx}"],
  },
  {
    name: "Banned gamification vocabulary (English-loan)",
    pattern: /\b(Unlocked!|Achievement|Streak|Level up|legendary|epic|mythic|rarity)\b/i,
    paths: ["src/components/badges/**/*.{ts,tsx}", "src/app/progress/**/*.{ts,tsx}", "src/app/api/reflections/**/*.{ts,tsx}", "src/app/api/program/**/*.{ts,tsx}"],
  },
  {
    name: "Banned animation libs on badges",
    pattern: /\b(canvas-confetti|lottie-react|lottie|tsparticles)\b/i,
    paths: ["src/components/badges/**/*.{ts,tsx}", "src/app/progress/**/*.{ts,tsx}"],
  },
  {
    name: "BADGE-09: no push notification on badge unlock",
    pattern: /\b(sendPush|push_notification|scheduleBadgeNotif)\b/i,
    paths: ["src/app/api/reflections/**/*.{ts,tsx}", "src/app/api/program/start-cycle/**/*.{ts,tsx}", "src/lib/badges/**/*.{ts,tsx}"],
  },
  {
    name: "No direct posthog.capture/track on sacred api paths (helper emitEvent is OK)",
    pattern: /posthog\.capture\(|\btrack\(/,
    paths: ["src/app/api/reflections/**/*.{ts,tsx}"],
  },
  {
    name: "No streak UI tokens",
    pattern: /🔥|streak_count|\bstreak\b/i,
    paths: ["src/components/badges/**/*.{ts,tsx}", "src/app/progress/**/*.{ts,tsx}"],
  },
  {
    name: "No progress-bar copy (3 of 7 unlocked)",
    pattern: /\d+\s+of\s+\d+\s+unlocked|\d+%\s+unlocked/i,
    paths: ["src/components/badges/**/*.{ts,tsx}", "src/app/progress/**/*.{ts,tsx}"],
  },
  // Special exception: `unlocked: boolean` is a structural TS prop name (Phase 7 decision —
  // 07.04-SUMMARY Rule 1 deviation). The vocabulary ban is on user-facing strings like
  // `Unlocked!` with punctuation, NOT on prop names. Above pattern uses `Unlocked!` exactly.
];

let failed = false;
for (const rule of RULES) {
  const files = rule.paths.flatMap((p) => globSync(p, { nodir: true }));
  for (const file of files) {
    const content = fs.readFileSync(file, "utf8");
    const lines = content.split("\n");
    lines.forEach((line, idx) => {
      if (rule.pattern.test(line)) {
        console.error(
          `[phase-8-guard] ${file}:${idx + 1}\n  Rule: ${rule.name}\n  Line: ${line.trim()}\n`,
        );
        failed = true;
      }
    });
  }
}

if (failed) {
  console.error("\n[phase-8-guard] FAILED — fix the above violations before release.\n");
  process.exit(1);
}
console.log("[phase-8-guard] PASS — no Phase 8 anti-patterns detected.");
process.exit(0);
```
