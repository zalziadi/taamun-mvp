**Shape**:

```typescript
"use client";

import { useEffect, useState, useMemo } from "react";
import { motion } from "framer-motion";
import { MilestoneBadge, type BadgeCode } from "@/components/badges/MilestoneBadge";
import { createBrowserClient } from "@supabase/ssr";

// Ordered for display: chronological within a cycle, cycle_complete last.
const BADGE_CODES_IN_ORDER: readonly BadgeCode[] = [
  "day_1", "day_3", "day_7", "day_14", "day_21", "day_28", "cycle_complete",
];

type BadgeRow = {
  badge_code: BadgeCode;
  cycle_number: number;
  unlocked_at: string;
};

export function BadgeGrid() {
  const [rows, setRows] = useState<BadgeRow[]>([]);
  const [status, setStatus] = useState<"loading" | "ok" | "empty" | "error">("loading");

  useEffect(() => {
    const supabase = createBrowserClient(
      process.env.NEXT_PUBLIC_SUPABASE_URL!,
      process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY!,
    );
    (async () => {
      const { data: authData } = await supabase.auth.getUser();
      if (!authData.user) {
        setStatus("empty");
        return;
      }
      const { data, error } = await supabase
        .from("badges")
        .select("badge_code, cycle_number, unlocked_at")
        .eq("user_id", authData.user.id)
        .order("cycle_number", { ascending: true });
      if (error) {
        setStatus("error");
        return;
      }
      const parsed: BadgeRow[] = (data ?? [])
        .filter((r): r is BadgeRow =>
          typeof r.badge_code === "string" &&
          BADGE_CODES_IN_ORDER.includes(r.badge_code as BadgeCode) &&
          typeof r.cycle_number === "number",
        )
        .map((r) => ({
          badge_code: r.badge_code as BadgeCode,
          cycle_number: r.cycle_number,
          unlocked_at: r.unlocked_at,
        }));
      setRows(parsed);
      setStatus(parsed.length > 0 ? "ok" : "empty");
    })();
  }, []);

  // Determine how many cycles to render rows for.
  //   Show cycles [1 .. max(3, max(cycle_number from rows))] — cap at 3 per BADGE-01.
  //   If user has no badges yet, show only cycle 1 (their in-progress cycle).
  const maxCycle = useMemo(() => {
    const max = rows.reduce((acc, r) => Math.max(acc, r.cycle_number), 1);
    return Math.min(3, Math.max(1, max));
  }, [rows]);

  const unlockedMap = useMemo(() => {
    const map = new Map<string, BadgeRow>();
    rows.forEach((r) => map.set(`${r.badge_code}:${r.cycle_number}`, r));
    return map;
  }, [rows]);

  if (status === "loading") {
    return (
      <div className="rounded-2xl border border-white/10 bg-white/5 p-6 animate-pulse">
        <div className="h-4 w-24 bg-white/10 rounded mb-4" />
        <div className="grid grid-cols-7 gap-3">
          {Array.from({ length: 7 }).map((_, i) => (
            <div key={i} className="aspect-square rounded-full bg-white/5" />
          ))}
        </div>
      </div>
    );
  }

  if (status === "error") return null; // silent — don't interrupt the page
  if (status === "empty") return null; // guest or no data — handled upstream by ProgressPage

  return (
    <section
      dir="rtl"
      className="rounded-2xl border border-[#c9b88a]/20 bg-white/5 p-6 space-y-6"
      aria-label="شارات الرحلة"
    >
      <div className="flex items-baseline justify-between">
        <h2 className="text-lg font-bold text-[#e8e1d9]">شارات الرحلة</h2>
        <span className="text-xs text-[#c9b88a]/60">خاصة بك — لا تُشارَك</span>
      </div>

      {Array.from({ length: maxCycle }).map((_, idx) => {
        const cycleNumber = idx + 1;
        return (
          <div key={cycleNumber} className="space-y-3">
            <h3 className="text-sm text-[#c9b88a]/80">
              الحلقة {toEasternArabic(cycleNumber)}
            </h3>
            <div className="grid grid-cols-4 gap-4 sm:grid-cols-7">
              {BADGE_CODES_IN_ORDER.map((code) => {
                const row = unlockedMap.get(`${code}:${cycleNumber}`);
                const unlocked = !!row;
                const tooltip = row
                  ? `${arabicLabel(code)} — ${formatArabicDate(row.unlocked_at)}`
                  : arabicLabel(code);
                return (
                  <motion.div
                    key={code}
                    initial={unlocked ? { opacity: 0 } : false}
                    animate={{ opacity: unlocked ? 1 : 0.4 }}
                    transition={{ duration: 0.6, ease: "easeOut" }}
                    title={tooltip}
                  >
                    <MilestoneBadge code={code} unlocked={unlocked} />
                  </motion.div>
                );
              })}
            </div>
          </div>
        );
      })}
    </section>
  );
}

function toEasternArabic(n: number): string {
  return n.toString().replace(/\d/g, (d) => "٠١٢٣٤٥٦٧٨٩"[Number(d)]);
}

function arabicLabel(code: BadgeCode): string {
  // Minimal label for tooltip; full names live in MilestoneBadge's BADGE_COPY.
  const labels: Record<BadgeCode, string> = {
    day_1: "اليوم ١",
    day_3: "اليوم ٣",
    day_7: "اليوم ٧",
    day_14: "اليوم ١٤",
    day_21: "اليوم ٢١",
    day_28: "اليوم ٢٨",
    cycle_complete: "تمام الحلقة",
  };
  return labels[code];
}

function formatArabicDate(iso: string): string {
  try {
    const d = new Date(iso);
    return new Intl.DateTimeFormat("ar-SA-u-nu-arab", {
      year: "numeric",
      month: "long",
      day: "numeric",
    }).format(d);
  } catch {
    return "";
  }
}
```

**Key constraints**:

1. **No modal, no dialog, no Sheet, no Drawer** — on click, nothing opens. The `title={tooltip}` attribute provides the subtle hover/long-press tooltip per BADGE-09. Do NOT wrap badges in `<button onClick>` or `<Link>`.

2. **No share button, no next/og import, no OpenGraph export** — BADGE-04. The header copy explicitly says "خاصة بك — لا تُشارَك" (private to you — not shared) to reinforce the tone.

3. **framer-motion usage is MINIMAL**:
   - Only `initial={{ opacity: 0 }} → animate={{ opacity: 1 }}` on unlocked badges.
   - `transition={{ duration: 0.6, ease: "easeOut" }}` — ≤ 600ms (08-CONTEXT).
   - NO particles, NO scale bursts, NO rotate, NO spring bounces.
   - Locked badges get `initial={false}` (no animation) and animate to opacity 0.4 — keeps them static in the grid.

4. **RLS-driven client read is sanctioned**: the Phase 7 migration's `users_select_own_badges` policy ensures `supabase.from("badges").select(...).eq("user_id", user.id)` only returns the user's own rows. No server-side API route needed. This keeps the page fast (NFR-01 LCP budget).

5. **Graceful fallback**:
   - `loading`: skeleton grid (7 circles animating).
   - `error`: render `null` (never interrupt the page; Plan 08.06 tests confirm the page is still navigable if the fetch fails).
   - `empty` (no user OR no badges yet): render `null` — the user is a guest, or a brand-new user who hasn't even completed Day 1 yet. Plan 08.02's fire-and-forget will create the first row on their Day-1 save; they see it next visit.

6. **Tooltip only on hover/long-press — no click handler**. The HTML `title` attribute degrades gracefully on mobile (iOS shows on long-press, Android shows on long-press). No custom tooltip library.

7. **Eastern Arabic numerals** in the cycle label "الحلقة ١" (matches `NFR-07` + 08-CONTEXT + Phase 11 YIR pattern). The date formatting uses `ar-SA-u-nu-arab` for a consistent calendar display.

8. **No PostHog capture in this component**. The badge grid is read-only display. `badge_unlock` fires server-side when the badge is created (helper). No `view_badges` or `badges_grid_opened` event in v1.2 — add later if measurement is needed, NOT in this plan.

9. **A11y**:
   - `aria-label="شارات الرحلة"` on the section.
   - `title={tooltip}` on each badge's motion wrapper for screen reader announce.
   - MilestoneBadge already has `role="img" aria-label={copy.name}` from Phase 7.
   - No interactive roles (no `tabIndex={0}`, no `role="button"`) because badges are not interactive.

- Import any new library (framer-motion is already installed; check `package.json` before writing the import).
