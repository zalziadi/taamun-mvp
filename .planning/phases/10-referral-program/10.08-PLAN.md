Skeleton:
```bash
#!/usr/bin/env bash
set -euo pipefail

FAIL=0

# Comment-line carve-out — strip JSDoc/line comments so docs aren't violations.
CARVE='grep -v ":[[:space:]]*\*\|:[[:space:]]*//"'

# ─── Bucket A: ReferralPanel + helpers ────────────────────────────────
A_PATHS="src/components/ReferralPanel.tsx src/components/ReferralPanelHelpers.ts"

# Vocab ban
if grep -rniE "\\b(earn|cash|reward\\b|points|affiliate|commission|أرباح|سحب|خصم)\\b" $A_PATHS 2>/dev/null | eval $CARVE | grep -q .; then
  echo "FAIL [A-vocab]: banned affiliate/earn/cash vocabulary in ReferralPanel"
  FAIL=1
fi

# Gamification ban
if grep -rniE "🔥|🏆|🥇|top.referrer|leaderboard|\\blevel.[0-9]" $A_PATHS 2>/dev/null | eval $CARVE | grep -q .; then
  echo "FAIL [A-gami]: gamification UI in ReferralPanel"
  FAIL=1
fi

# Timers / modals / animations
if grep -rniE "setInterval|\\bcountdown\\b|<Dialog|<Modal|@radix-ui/react-dialog|fixed inset-0|framer-motion" $A_PATHS 2>/dev/null | eval $CARVE | grep -q .; then
  echo "FAIL [A-ui]: banned timer/modal/animation usage"
  FAIL=1
fi

# REQUIRED Arabic copy (da'wah framing — REFER-10)
if ! grep -q "ادعُ للتمعّن" src/components/ReferralPanel.tsx; then
  echo "FAIL [A-copy]: missing required Arabic da'wah string 'ادعُ للتمعّن'"
  FAIL=1
fi

# ─── Bucket B: /api/referral/** + activate route (analytics prefix-only) ─
B_PATHS="src/app/api/referral src/app/api/activate/route.ts"

# Full-code leak to analytics
if grep -rnE "emitEvent[^)]*code[^)]*\\)|properties:[^}]*code:[[:space:]]*code\\b" $B_PATHS 2>/dev/null | eval $CARVE | grep -q .; then
  echo "FAIL [B-analytics]: full code may be leaked to analytics (ANALYTICS-07 violation)"
  FAIL=1
fi

# ─── Bucket C: credit paths (REFER-04 / REFER-05 / REFER-11) ──────────
C_PATHS="src/lib/referral/credit.ts src/app/api/cron/credit-referrals"

# REFER-04: NEVER mint activation_codes from credit path
if grep -rnE 'from\("activation_codes"\).*insert|insert.*activation_codes' $C_PATHS 2>/dev/null | eval $CARVE | grep -q .; then
  echo "FAIL [C-mint]: credit path appears to mint activation_codes (REFER-04 violation)"
  FAIL=1
fi

# REFER-05: synchronous only — no webhooks / queues from credit
if grep -rniE "webhook|queue\\.add|enqueue" $C_PATHS 2>/dev/null | eval $CARVE | grep -q .; then
  echo "FAIL [C-async]: credit path contains async dispatch (REFER-05 violation)"
  FAIL=1
fi

# REFER-11: silent credit — no email/push/emitEvent tied to credit
if grep -rniE "email_queue|push_queue|emitEvent.*credited|emitEvent.*\\breward\\b" $C_PATHS 2>/dev/null | eval $CARVE | grep -q .; then
  echo "FAIL [C-notify]: credit path emits notification/event (REFER-11 violation)"
  FAIL=1
fi

# ─── Bucket D: /account/referral page — no client analytics for referral events ─
D_PATHS="src/app/account/referral"

if grep -rniE "posthog\\.capture|analytics\\.track\\(.*referral" $D_PATHS 2>/dev/null | eval $CARVE | grep -q .; then
  echo "FAIL [D-client-analytics]: client-side analytics in /account/referral (ANALYTICS-07)"
  FAIL=1
fi

# ─── Bucket E: auto-share ban (SUMMARY §R4) ───────────────────────────
E_PATHS="src/components/ReferralPanel.tsx src/app/account/referral"

if grep -rniE "useEffect[^}]*\\.click\\(|ref\\.current\\?\\.click\\(" $E_PATHS 2>/dev/null | eval $CARVE | grep -q .; then
  echo "FAIL [E-autoshare]: auto-share pattern detected (SUMMARY §R4 violation)"
  FAIL=1
fi

if [ "$FAIL" -eq 0 ]; then
  echo "phase-10 guard: PASS"
  exit 0
else
  echo "phase-10 guard: FAIL"
  exit 1
fi
```
