**src/lib/referral/generate.ts** — exact exports:

```ts
// imports
import { randomBytes } from "crypto";
import type { SupabaseClient } from "@supabase/supabase-js";

// constants
export const FRIEND_PREFIX = "FRIEND-" as const;
// Crockford base32 minus I, L, O, U (confusable with 1 / 0 / V).
const CROCKFORD_CHARS = "0123456789ABCDEFGHJKMNPQRSTVWXYZ"; // length 32
// Regex guards Phase 10.04 detection of FRIEND-* vs TAAMUN-* at activation time.
export const FRIEND_CODE_REGEX = /^FRIEND-[0-9A-HJKMNP-TV-Z]{6}$/;

// pure generator: 6 random chars from the 32-char alphabet
export function generateFriendCode(): string { /* crypto.randomBytes + modulo */ }

// collision-aware: queries public.referrals.code to guarantee uniqueness
export async function generateUniqueFriendCode(
  admin: SupabaseClient,
  maxRetries = 5,
): Promise<string> {
  for (let i = 0; i < maxRetries; i++) {
    const candidate = generateFriendCode();
    const { data } = await admin
      .from("referrals")
      .select("id")
      .eq("code", candidate)
      .maybeSingle();
    if (!data) return candidate;
  }
  throw new Error("referral_code_collision_exhausted");
}
```

Notes:
- `generateFriendCode()` internal: read `randomBytes(6)` and, for each byte, index into `CROCKFORD_CHARS` via `byte % 32`. Modulo 32 is uniform because 256 is a multiple of 32 (no bias).
- Use `SupabaseClient` type imported from `@supabase/supabase-js` (already in the project's dependency tree — verify via `grep "@supabase/supabase-js" package.json`).
- Do NOT bring in `nanoid` / `uuid` / any new dep (NFR-08).
