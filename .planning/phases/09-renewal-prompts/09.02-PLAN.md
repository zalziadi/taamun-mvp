Template (executor — adapt column names if schema introspection on staging reveals different identifiers; the three tables named in 09-CONTEXT are all referenced by `user_id` except `activation_codes` which uses `used_by`):

```sql
-- v1.2 Phase 9: Renewal Prompts In-App — Step 2 of 2 (backfill)
-- Populates profiles.original_gateway for existing users by reading from the
-- four source-of-truth tables. See .planning/phases/09-renewal-prompts/09.02-PLAN.md
-- for precedence rationale.
--
-- Idempotent: every UPDATE is gated by `WHERE original_gateway IS NULL`, so
-- re-applying the migration affects zero rows on the second run.

-- ── 1. Salla subscribers ─────────────────────────────────────────────────────
UPDATE public.profiles p
SET original_gateway = 'salla'
WHERE p.original_gateway IS NULL
  AND EXISTS (
    SELECT 1 FROM public.salla_integration s WHERE s.user_id = p.id
  );

-- ── 2. Tap subscribers (only users not already tagged Salla) ─────────────────
UPDATE public.profiles p
SET original_gateway = 'tap'
WHERE p.original_gateway IS NULL
  AND EXISTS (
    SELECT 1 FROM public.tap_customer_subscriptions t WHERE t.user_id = p.id
  );

-- ── 3. Stripe subscribers (only users not already tagged) ────────────────────
UPDATE public.profiles p
SET original_gateway = 'stripe'
WHERE p.original_gateway IS NULL
  AND EXISTS (
    SELECT 1 FROM public.customer_subscriptions c WHERE c.user_id = p.id
  );

-- ── 4. Eid-campaign activation-code redeemers (users with no gateway row) ────
UPDATE public.profiles p
SET original_gateway = 'eid_code'
WHERE p.original_gateway IS NULL
  AND EXISTS (
    SELECT 1 FROM public.activation_codes a
     WHERE a.used_by = p.id
  );

-- Users whose original_gateway remains NULL after this migration are grandfathered
-- with no payment record (manual grants, admin accounts, etc.). RenewalBanner
-- (Plan 09.04) treats NULL as "do not show banner" — per 09-CONTEXT §Claude's
-- Discretion recommendation.
```
