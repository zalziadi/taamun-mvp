// ── Fake Supabase client — deterministic, in-memory ───────────────────────────
function makeFakeSupabase(fixtures) {
  return {
    from(table) {
      const rows = fixtures[table] || [];
      const chain = {
        _rows: rows,
        _filters: [],
        select() { return chain; },
        eq(col, val) { chain._filters.push(r => r[col] === val); return chain; },
        in(col, vals) { chain._filters.push(r => vals.includes(r[col])); return chain; },
        gte(col, val) { chain._filters.push(r => r[col] >= val); return chain; },
        is(col, val) {
          if (val === null) chain._filters.push(r => r[col] === null || r[col] === undefined);
          return chain;
        },
        limit() { return chain; },
        maybeSingle: async () => {
          const matched = chain._rows.filter(r => chain._filters.every(f => f(r)));
          return { data: matched[0] ?? null, error: null };
        },
        async then(resolve) { // make it thenable for .select() without .maybeSingle
          const matched = chain._rows.filter(r => chain._filters.every(f => f(r)));
          resolve({ data: matched, error: null });
        },
      };
      return chain;
    },
  };
}
