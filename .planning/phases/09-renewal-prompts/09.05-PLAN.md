```ts
import { getSupabaseAdmin } from "@/lib/supabaseAdmin";

export type ShouldShowResult =
  | {
      show: false;
      reason:
        | "no_profile"
        | "no_expires_at"
        | "not_within_window"
        | "already_expired"
        | "no_gateway"
        | "email_sent_today"
        | "push_sent_today"
        | "db_error";
    }
  | {
      show: true;
      gateway: "salla" | "tap" | "stripe" | "eid_code";
      daysRemaining: number;
      tier: string;
    };

const SEVEN_DAYS_MS = 7 * 24 * 60 * 60 * 1000;
const ONE_DAY_MS = 24 * 60 * 60 * 1000;

/**
 * Server-side authority for "should the renewal banner render for this user right now?"
 * Consolidates the 7-day window check, auto-renewed suppression, and 3-channel
 * nudge dedup (email + push + banner) from REQUIREMENTS RENEW-01, RENEW-06, RENEW-07.
 * RENEW-08: never originates an email — only READS email_queue to avoid duplicate
 * nudging the same day (reuses the existing expiry-warning-template.ts-driven queue).
 */
export async function shouldShowRenewalBanner(
  userId: string
): Promise<ShouldShowResult> {
  let admin;
  try {
    admin = getSupabaseAdmin();
  } catch {
    return { show: false, reason: "db_error" };
  }

  // 1. Profile lookup
  const { data: profile, error: profileError } = await admin
    .from("profiles")
    .select("expires_at, original_gateway, subscription_tier")
    .eq("id", userId)
    .maybeSingle();

  if (profileError) {
    console.warn("[shouldShow] profile read error:", profileError.message);
    return { show: false, reason: "db_error" };
  }
  if (!profile) return { show: false, reason: "no_profile" };
  if (!profile.expires_at) return { show: false, reason: "no_expires_at" };

  const expiresAtMs = new Date(profile.expires_at).getTime();
  const now = Date.now();

  if (expiresAtMs <= now) return { show: false, reason: "already_expired" };
  if (expiresAtMs - now > SEVEN_DAYS_MS) {
    // Auto-renewed users land here naturally: their expires_at jumps well past 7 days.
    return { show: false, reason: "not_within_window" };
  }
  if (!profile.original_gateway) return { show: false, reason: "no_gateway" };

  // 2. Email dedup — respect the existing v1.1 expiry-warning queue
  const since = new Date(now - ONE_DAY_MS).toISOString();
  const { data: recentEmails } = await admin
    .from("email_queue")
    .select("id")
    .eq("user_id", userId)
    .in("type", ["expiry_warning", "renewal"])
    .gte("sent_at", since)
    .limit(1);
  if (recentEmails && recentEmails.length > 0) {
    return { show: false, reason: "email_sent_today" };
  }

  // 3. Push dedup — if user received any push in the last 24h, skip banner
  const { data: pushRows } = await admin
    .from("push_subscriptions")
    .select("last_notified_at")
    .eq("user_id", userId)
    .gte("last_notified_at", since)
    .limit(1);
  if (pushRows && pushRows.length > 0) {
    return { show: false, reason: "push_sent_today" };
  }

  // 4. Compute days remaining (ceil — "6.3 days" reads as "6 more days")
  const daysRemaining = Math.max(
    0,
    Math.ceil((expiresAtMs - now) / (24 * 60 * 60 * 1000))
  );

  // Narrow gateway to the expected union; if a rogue value sneaks in, treat as no_gateway
  const gateway = profile.original_gateway;
  if (
    gateway !== "salla" &&
    gateway !== "tap" &&
    gateway !== "stripe" &&
    gateway !== "eid_code"
  ) {
    return { show: false, reason: "no_gateway" };
  }

  return {
    show: true,
    gateway,
    daysRemaining,
    tier: profile.subscription_tier || "monthly",
  };
}
```
