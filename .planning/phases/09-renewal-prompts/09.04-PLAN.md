```ts
import { NextResponse } from "next/server";
import { requireUser } from "@/lib/authz";
import { shouldShowRenewalBanner } from "@/lib/renewal/shouldShow";
import { refreshEntitlementIfStale } from "@/lib/renewal/refreshEntitlement";

export const dynamic = "force-dynamic";
export const runtime = "nodejs";

export async function GET() {
  const auth = await requireUser();
  if (!auth.ok) {
    // Unauthenticated → banner should never show
    return NextResponse.json({ show: false, reason: "unauthenticated" });
  }

  // Plan 09.06 contract: refresh cookie if DB shows a later expiry than cookie.
  // This runs BEFORE shouldShowRenewalBanner reads expires_at so the banner
  // never shows to a user who just renewed but has a stale cookie (Pitfall 14).
  const res = NextResponse.json({ show: false }); // placeholder we mutate below
  try {
    await refreshEntitlementIfStale(auth.user.id, res);
  } catch (e) {
    console.warn("[renewal/status] entitlement refresh failed (non-blocking):", e);
  }

  const result = await shouldShowRenewalBanner(auth.user.id);

  // Re-create the response carrying any Set-Cookie the refresh may have attached.
  const body = result.show
    ? {
        show: true,
        gateway: result.gateway,
        daysRemaining: result.daysRemaining,
        tier: result.tier,
      }
    : { show: false, reason: result.reason };

  const final = NextResponse.json(body);
  // Propagate any cookie the refresh may have set onto the placeholder response.
  res.cookies.getAll().forEach((c) => final.cookies.set(c));
  return final;
}
```

```tsx
"use client";

import { useEffect, useState, useCallback } from "react";
import { usePathname } from "next/navigation";
import { isExcludedPath } from "@/lib/analytics/excludedPaths";

const DISMISS_KEY = "taamun.renewal_dismissed_until.v1";
const DISMISS_COUNT_KEY = "taamun.renewal_dismiss_count.v1";
const SESSION_EMIT_KEY = "taamun.renewal_prompted_session.v1";
const DISMISS_WINDOW_MS = 48 * 60 * 60 * 1000; // 48h
const MAX_DISMISSALS = 3;

type Status =
  | { show: false; reason?: string }
  | { show: true; gateway: "salla" | "tap" | "stripe" | "eid_code"; daysRemaining: number; tier: string };

function isDismissed(): boolean {
  if (typeof window === "undefined") return false;
  try {
    const until = Number(window.localStorage.getItem(DISMISS_KEY) || 0);
    if (Number.isFinite(until) && until > Date.now()) return true;
    const count = Number(window.localStorage.getItem(DISMISS_COUNT_KEY) || 0);
    if (count >= MAX_DISMISSALS) return true;
  } catch {
    // SSR or storage blocked → treat as not-dismissed, component will no-op later if fetch fails
  }
  return false;
}

function gatewayCtaHref(gateway: Status extends { show: true; gateway: infer G } ? G : never): string {
  switch (gateway) {
    case "salla":
      return process.env.NEXT_PUBLIC_SALLA_RENEWAL_URL || "/pricing?source=expired&gateway=salla";
    case "tap":
      return process.env.NEXT_PUBLIC_TAP_RENEWAL_URL || "/pricing?source=expired&gateway=tap";
    case "stripe":
      return process.env.NEXT_PUBLIC_STRIPE_PORTAL_URL || "/pricing?source=expired&gateway=stripe";
    case "eid_code":
    default:
      return "/pricing?source=expired";
  }
}

export function RenewalBanner() {
  const pathname = usePathname();
  const [status, setStatus] = useState<Status>({ show: false });
  const [hidden, setHidden] = useState(false);

  // Sacred-path structural guard — runs BEFORE any fetch
  const onSacredPath = pathname ? isExcludedPath(pathname) : true;

  useEffect(() => {
    if (onSacredPath) return;
    if (isDismissed()) return;
    let cancelled = false;
    (async () => {
      try {
        const r = await fetch("/api/renewal/status", { cache: "no-store" });
        if (!r.ok) return;
        const data: Status = await r.json();
        if (!cancelled) setStatus(data);
      } catch {
        // Network error → silently no-op, never crash the page
      }
    })();
    return () => {
      cancelled = true;
    };
  }, [onSacredPath, pathname]);

  // Fire renewal_prompted exactly once per session when banner first becomes visible
  useEffect(() => {
    if (!status.show || hidden || onSacredPath) return;
    if (typeof window === "undefined") return;
    try {
      if (window.sessionStorage.getItem(SESSION_EMIT_KEY) === "1") return;
      if (status.gateway === "eid_code") {
        // Skip analytics for eid_code path (not in TypedEvent gateway union). See plan note.
        window.sessionStorage.setItem(SESSION_EMIT_KEY, "1");
        return;
      }
      // Delegate to the existing client analytics wrapper. See src/lib/analytics.ts `track()`.
      import("@/lib/analytics").then(({ track }) => {
        track("renewal_prompted", {
          renewal_days_remaining: status.daysRemaining,
          gateway: status.gateway,
          tier: status.tier,
        });
      });
      window.sessionStorage.setItem(SESSION_EMIT_KEY, "1");
    } catch {
      /* sessionStorage blocked → do nothing */
    }
  }, [status, hidden, onSacredPath]);

  const handleDismiss = useCallback(() => {
    try {
      const until = Date.now() + DISMISS_WINDOW_MS;
      window.localStorage.setItem(DISMISS_KEY, String(until));
      const prev = Number(window.localStorage.getItem(DISMISS_COUNT_KEY) || 0);
      window.localStorage.setItem(DISMISS_COUNT_KEY, String(prev + 1));
    } catch {
      /* ignore */
    }
    setHidden(true);
  }, []);

  if (onSacredPath) return null;
  if (!status.show) return null;
  if (hidden) return null;

  const href = gatewayCtaHref(status.gateway);

  return (
    <div
      role="status"
      dir="rtl"
      className="border-b border-[#2A2621] bg-[#14120F] px-4 py-3 text-sm text-[#D6D1C8]"
    >
      <div className="mx-auto flex max-w-[1240px] items-center justify-between gap-3">
        <p className="flex-1">
          {/* Copy uses "واصل" framing — no scarcity, no "لا تفقد". */}
          واصل معنا — اشتراكك يقترب من الانتهاء.
        </p>
        <a
          href={href}
          className="rounded-md border border-[#C9A84C] px-3 py-1 text-[#C9A84C] transition-opacity hover:opacity-80"
        >
          واصل رحلتك
        </a>
        <button
          type="button"
          onClick={handleDismiss}
          aria-label="إخفاء التذكير"
          className="rounded-md px-2 py-1 text-[#807A72] outline-none focus-visible:ring-2 focus-visible:ring-[#C9A84C]/45"
        >
          ✕
        </button>
      </div>
    </div>
  );
}
```
