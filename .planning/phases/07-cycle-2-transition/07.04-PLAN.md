Structure (matches existing `/api/program/start-cycle/route.ts` auth + admin pattern):

```typescript
import { NextResponse } from "next/server";
import { createSupabaseServerClient } from "@/lib/supabase/server";
import { getSupabaseAdmin } from "@/lib/supabaseAdmin";
import { emitEvent } from "@/lib/analytics/server";
import type { TypedEvent } from "@/lib/analytics/events";

const PHASE_7_ALLOWED_BADGE_CODES = new Set(["day_28"]); // Phase 8 will extend

export async function POST(req: Request) {
  // 1. Auth (identical pattern to start-cycle)
  const supabase = await createSupabaseServerClient();
  const { data: auth, error: authErr } = await supabase.auth.getUser();
  if (authErr || !auth.user) {
    return NextResponse.json({ error: "unauthorized" }, { status: 401 });
  }

  // 2. Parse + validate body
  const body = (await req.json().catch(() => ({}))) as {
    badge_code?: string;
    cycle_number?: number;
    day_number?: number;
  };

  const badge_code = body.badge_code;
  const cycle_number = body.cycle_number;
  const day_number = body.day_number;

  if (!badge_code || typeof cycle_number !== "number" || typeof day_number !== "number") {
    return NextResponse.json({ error: "invalid_body" }, { status: 400 });
  }
  if (!PHASE_7_ALLOWED_BADGE_CODES.has(badge_code)) {
    return NextResponse.json(
      { error: "badge_code_not_in_phase_7_scope", allowed: Array.from(PHASE_7_ALLOWED_BADGE_CODES) },
      { status: 400 }
    );
  }

  // 3. Idempotent insert via admin client (RLS bypass per Plan 07.01)
  const admin = getSupabaseAdmin();
  const userId = auth.user.id;

  // Upsert with ON CONFLICT DO NOTHING pattern — Supabase-js: .upsert(..., { onConflict, ignoreDuplicates: true })
  // The UNIQUE constraint on (user_id, badge_code, cycle_number) is the load-bearing guard.
  const { data, error } = await admin
    .from("badges")
    .upsert(
      {
        user_id: userId,
        badge_code,
        cycle_number,
        unlocked_at: new Date().toISOString(),
        notified: true, // Plan 07.04 unlocks happen with the CTA tap → user is in the moment → notified semantics: already surfaced
      },
      { onConflict: "user_id,badge_code,cycle_number", ignoreDuplicates: true }
    )
    .select(); // .select() forces returning rows — length 0 means a duplicate (skipped) insert

  if (error) {
    return NextResponse.json({ error: "db_error", details: error.message }, { status: 500 });
  }

  const actuallyInserted = Array.isArray(data) && data.length > 0;

  // 4. Emit event ONLY if a row was actually inserted — enforces RETURN-07 "exactly once"
  if (actuallyInserted) {
    const event: TypedEvent = {
      name: "badge_unlock",
      properties: { badge_code, day_number, cycle_number },
    };
    await emitEvent(event, userId); // fire-and-forget
  }

  return NextResponse.json({
    ok: true,
    unlocked: actuallyInserted,
    reason: actuallyInserted ? undefined : "already_unlocked",
  });
}
```
